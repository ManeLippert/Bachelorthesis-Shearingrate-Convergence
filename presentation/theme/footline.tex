  \end{beamercolorbox}%
  \begin{beamercolorbox}[wd=.1\paperwidth,ht=2.25ex,dp=1.0ex,right,rightskip=0.3cm]{section in head/foot}%
    \insertframenumber{}
  \end{beamercolorbox}}%
  \vskip0pt%
}
